#include<stdlib.h>

struct SNode{
        char*top;
        char*base;
};

typedef struct SNode * SqStack;

SqStack InitStack(){
        SqStack S=(SqStack)malloc(sizeof(struct SNode));
        S->base=(char*)malloc(100*sizeof(char));
        S->top=S->base;
        return S;
}

void push(SqStack S,int number){
        *(S->top)=number;
        (S->top)++;
}

int pop(SqStack S){
        int number;
        (S->top)--;
        number=*(S->top);
        return number;
}

        while(c1!=0){

                c1--;
